# üöÄ - —Ä–∞—Å—Å—á–∏—Ç–∞–π —Ç—è–≥—É —Ä–∞–∫–µ—Ç–Ω–æ–≥–æ –¥–≤–∏–≥–∞—Ç–µ–ª—è –Ω–∞ –æ—Å–Ω–æ–≤–µ –≤–≤–µ–¥–µ–Ω–Ω—ã—Ö –∑–Ω–∞—á–µ–Ω–∏–π  

```python
import math

def calculate_thrust(m_dot, T_c, gamma, M, P_e, P_c, A_e, P_a):
    """
    –†–∞—Å—á–µ—Ç —Ç—è–≥–∏ —Ä–∞–∫–µ—Ç–Ω–æ–≥–æ –¥–≤–∏–≥–∞—Ç–µ–ª—è.
    """
    # –£–Ω–∏–≤–µ—Ä—Å–∞–ª—å–Ω–∞—è –≥–∞–∑–æ–≤–∞—è –ø–æ—Å—Ç–æ—è–Ω–Ω–∞—è (–î–∂/–º–æ–ª—å¬∑–ö)
    R = 8.314

    # –†–∞—Å—á–µ—Ç —ç—Ñ—Ñ–µ–∫—Ç–∏–≤–Ω–æ–π —Å–∫–æ—Ä–æ—Å—Ç–∏ –∏—Å—Ç–µ—á–µ–Ω–∏—è v_e
    v_e = math.sqrt(
        (2 * gamma * R * T_c) / ((gamma - 1) * M) * (1 - (P_e / P_c) ** ((gamma - 1) / gamma))
    )

    # –†–∞—Å—á–µ—Ç —Ç—è–≥–∏
    thrust = m_dot * v_e + (P_e - P_a) * A_e

    # –í—ã–≤–æ–¥ —Ä–µ–∑—É–ª—å—Ç–∞—Ç–∞
    print(f"–¢—è–≥–∞ –¥–≤–∏–≥–∞—Ç–µ–ª—è: {thrust:.2f} –ù")

# –í–≤–æ–¥ –¥–∞–Ω–Ω—ã—Ö –æ—Ç –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—è
m_dot = float(input('–í–≤–µ–¥–∏—Ç–µ –º–∞—Å—Å–æ–≤—ã–π —Ä–∞—Å—Ö–æ–¥ —Ç–æ–ø–ª–∏–≤–∞ (–∫–≥/—Å): '))
T_c = float(input('–í–≤–µ–¥–∏—Ç–µ —Ç–µ–º–ø–µ—Ä–∞—Ç—É—Ä—É –≤ –∫–∞–º–µ—Ä–µ —Å–≥–æ—Ä–∞–Ω–∏—è (–ö): '))
gamma = float(input('–í–≤–µ–¥–∏—Ç–µ –ø–æ–∫–∞–∑–∞—Ç–µ–ª—å –∞–¥–∏–∞–±–∞—Ç—ã: '))
M = float(input('–í–≤–µ–¥–∏—Ç–µ –º–æ–ª–µ–∫—É–ª—è—Ä–Ω—É—é –º–∞—Å—Å—É –ø—Ä–æ–¥—É–∫—Ç–æ–≤ —Å–≥–æ—Ä–∞–Ω–∏—è (–∫–≥/–º–æ–ª—å): '))
P_e = float(input('–í–≤–µ–¥–∏—Ç–µ –¥–∞–≤–ª–µ–Ω–∏–µ –Ω–∞ –≤—ã—Ö–æ–¥–µ –∏–∑ —Å–æ–ø–ª–∞ (–ü–∞): '))
P_c = float(input('–í–≤–µ–¥–∏—Ç–µ –¥–∞–≤–ª–µ–Ω–∏–µ –≤ –∫–∞–º–µ—Ä–µ —Å–≥–æ—Ä–∞–Ω–∏—è (–ü–∞): '))
A_e = float(input('–í–≤–µ–¥–∏—Ç–µ –ø–ª–æ—â–∞–¥—å –≤—ã—Ö–æ–¥–Ω–æ–≥–æ —Å–µ—á–µ–Ω–∏—è —Å–æ–ø–ª–∞ (–º^2): '))
P_a = float(input('–í–≤–µ–¥–∏—Ç–µ –∞—Ç–º–æ—Å—Ñ–µ—Ä–Ω–æ–µ –¥–∞–≤–ª–µ–Ω–∏–µ (–ü–∞): '))

# –í—ã–∑–æ–≤ —Ñ—É–Ω–∫—Ü–∏–∏ —Å –≤–≤–µ–¥–µ–Ω–Ω—ã–º–∏ –∑–Ω–∞—á–µ–Ω–∏—è–º–∏
calculate_thrust(m_dot, T_c, gamma, M, P_e, P_c, A_e, P_a)
```
